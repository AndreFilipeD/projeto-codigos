Ainda considerando o Person da parte 1 de Props, veja o exemplo
de como melhorar o código, utilizando essa nova estrutura grifada
em amarelo, onde na hora de adicionar os valores no JSX, não precisamos 
acessar o props.name, e sim apenas o name, entre outros ::

type Props = {
    name: string;
    avatar: string;
    roles: string[];
}

export const Person = (props: Props) => {
    
    const { name, avatar, roles } = props;
    //parte do exemplo mencionado

    return{
        <>

            <h1>name</h1>

            <img 
                src={avatar}
                alt={name}
                className="w-40" //css
            />

            <ul>
                <li>{roles[0]}</li>
                <li>{roles[1]}</li>
            </ul>
        
        </>
    }
}

:: Mas podemos simplificar essa estrutura ainda mais, colocando
esse const diretamente no export do componente conforme exemplo
abaixo ::

export const Person = ({name, avatar, roles}: Props) => {

    return{
        <>

            <h1>name</h1>

            <img 
                src={avatar}
                alt={name}
                className="w-40" //css
            />

            <ul>
                <li>{roles[0]}</li>
                <li>{roles[1]}</li>
            </ul>
        
        </>
    }
}
