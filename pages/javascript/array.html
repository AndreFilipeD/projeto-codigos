<SUBSTITUIÇÃO>

Considerando :: let listaNumero = [1,2,3,4,55,66,777];

listaNumero[0] = 88; listaNumero[1] = 99;

window.alert(listaNumero); :: Resultado: [88,99,3,4,55,66,777]
<COMANDOS>

listaNumero.push(2);
listaNumero.push(22);
listaNumero.push(222);

window.alert(listaNumero); :: Resultado: [88,99,3,4,55,66,777,2,22,222]
Push acrescenta sempre no final do Array

window.alert(listaNumero.length); :: Resultado: 10

window.alert(listaNumero[listaNumero.length - 1]); ::
Resultado: 222 :: mostra o ultimo valor da lista

listaNumero.pop(); :: remove ultimo item da lista :: 222

listaNumero.shift(); :: remove primeiro item da lista :: 88

listaNumero.unshift(345); :: adiciona 345 ao início da lista.

listaNumero.indexOf(66); :: retorna o numero 5, posição na lista

listaNumero.splice(2, 3); :: o primeiro valor é a posição inicial,
e removerá os 3 conteudos em direção ao fim :: removerá :: 3, 4, 55

var copiar = listaNumero.slice(); :: copia conteudo para outra lista
