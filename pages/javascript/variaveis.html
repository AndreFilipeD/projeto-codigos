<JS : VARIAVEIS> 
<DECLARAÇÃO>

var nome = "andre" :: typeof nome :: string
var salario = 0 :: typeof salario :: number
var sorte = false ou f :: typeof sorte :: boolean
<CONVERSÃO>

Number.parseInt(variavel) :: converte a variavel para inteiro.
Number.parseFloat(variavel) :: converte a variavel para float.

Number(variavel) :: JS se vira pra converter em numero.

variavel.toString() :: converte variavel numerica para string.

variavel.toLocateString('pt-BR',{style:'currency',currency:'BRL'}) ::
Utiliza o padrão monetario Brasileiro na apresentação do valor;
Use em 'BRL' o 'USD' para dolar, 'EUR' para euro entre outros...

<MANIPULAÇÃO>

variavel.toFixed(2) :: adiciona 2 casas depois do ponto.

variavel.replace('.',',') :: troca o ponto pela virgula
deve funcionar para qualquer caractere da variavel (testar);

´Eu estou aprendendo ${variavel}´ :: onde variavel = 'javascript'
Utilizar dessa forma precisa estar entre crases [`...`]

variavel.lenght :: informa a quantidade de caracteres

variavel.toUpperCase() :: torna os caracteres maiusculos

variavel.toLowerCase() :: torna os caracteres minusculos
