<CLONAGEM EM MASSA>

var shopCont = 0 
var shopItem=[]
var shopList = window.document.querySelector('#right')

function create(item){
    switch(item){
        case 0:
           shopCont++
           shopItem[shopCont] = window.document.createElement('div')
           shopItem[shopCont].setAttribute('class','shopj')
           shopItem[shopCont].innerHTML="item "+shopCont+" criado!!"
           shopList.appendChild(shopItem[shopCont])
           break;
       case 1:
           shopItem[shopCont].parentNode.removeChild(shopItem[shopCont])
           shopCont--
           break;
    }
}
