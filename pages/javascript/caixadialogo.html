<JS : CAIXAS DE DIALOGO> 
<ALERTA>

Utilize esse comando para criar um alerta.

window.alert('Olá, eu sou um Alerta!');
ou
alert('Olá, eu sou um Alerta!');

<CONFIRMAÇÃO>

Utilize a caixa de confirmação para receber
um valor Booleano True ou False ::

var r = confirm("Escolha um valor!");

if(r=true){
    alert('Voce pressionou OK!')
}else{
    alert('Voce pressionou Cancelar!')
}

<PROMPT DE ENTRADA>

var idade = prompt("Digite sua idade:");

if(idade!=null){
    alert("Idade: "+ idade +"anos.");
}
