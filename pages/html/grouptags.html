<HTML5 : GROUPING TAGS>
<GRUPOS PRINCIPAIS>

<header> :: cabeçalho
<main> :: principal
<footer> :: rodapé

<GRUPOS SECUNDARIOS>

<div> :: divisão vertical de conteudos :: block-level
<span> :: período horizontal de conteudos :: inline-level
<GRUPOS DISTINTOS>

<nav> :: navegação :: envelopa botões e dados de navegação

<article> :: envelopa elementos de um artigo ::
imagem, texto, titulo etc...

<aside> :: conteúdo que é tangencialmente relacionado
ao conteúdo do seu entorno.

<section> :: Seções com conteúdo da pagina

<fieldset> :: 
Agrupa um conjunto de entradas e titulos (input & labels).

<figure>:: Envelopa uma figura (imagem)

    <img src="data.png" alt="info">
    <figcaption> ... </figcaption> :: Texto/Legenda da figura envelopada.

</figure>
