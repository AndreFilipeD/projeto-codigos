<HTML5 : LEVEL TAGS> 

<BLOCK-LEVEL> :: Inicia em uma nova linha.

<header> :: cabeçalho
<main> :: principal
<footer> :: rodapé
<div> :: divisão
<section> :: seção
<article> :: artigo
<aside> :: á parte
<nav> :: navegação
<h1>-<h6> :: titulos
<details> :: Informação com conteudo ocultavel
<form> :: formulário
<video> :: vídeo
<adress> :: endereço
<blockquote> :: bloco de citação
<canvas> :: tela
<fieldset> :: conjunto de campos
<figcaption> :: capitura de figura
<figure> :: figura
<hr> :: cria linha horizontal :: horizontal row **
<noscript> :: age caso script no navegador = desativado
<pre> :: mantem os textos conforme digitados no arquivo
<dd> - <dl> - <dt> - <ol> - <ul> :: listas
<li> :: elemento na lista
<table> :: Inicialização de Tabela
<tfoot> :: Rodapé da Tabela
<tbody> :: Rodapé da Tabela
<thead> :: Cabeçalho da Tabela

<span> :: período, é um div inline level
<a> :: ancora
<b> :: bold - negrito
<br> :: quebra de linha
<bdo> :: invertido
<code> :: código
<cite> :: Citar um livro ou artista
<dfn> :: Define valor semantico a um termo
<i> :: Itálico
<img> :: Elemento imagem
<small> :: pequeno
<strong> :: bold
<em> :: êmfase
<sup> - <sub> :: formulas
<label> :: rótulo
<button> :: botão
<input> :: entrada de dados
<select> :: selecionar
<iframe> :: quadro em linha
<abbr> :: conteudo com informação ao tocar
<object> :: exibe conteudos externos ex: pdf
<meter> :: barra de carregamento
<progress> :: barra de progresso
<textarea> :: área que recebe muitos caracteres
<time> :: Semantica - conteudo tempo
<var> :: Semantica - variavel matemática :: x+1=2
