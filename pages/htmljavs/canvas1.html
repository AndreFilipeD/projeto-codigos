<INICIALIZAÇÃO JAVASCRIPT>

var canvas = window.document.getElementById("areaCanvas");
var ctx = canvas.getContext("2d");
::
Coletamos o elemento canvas pelo ID, e depois
extraimos o contexto 2D desse mesmo canvas para uma
variavel separada, assim atravez dessa variavel podemos
desenhar nesse canvas utilizando o contexto 2D.
<DESENHANDO LINHAS>

ctx.beginPath();
:: Inicializamos um novo caminho

ctx.moveTo(0, 0);
:: Posicionamos o ponteiro em 0x e 0y

ctx.lineTo(300, 150);
:: Movemos o ponteiro para 300x e 150y

ctx.stroke();
:: Comando para realizar o desenho no caminho já inicializado.
