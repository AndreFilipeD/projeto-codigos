<TYPESCRIPT : TIPAGEM>
<TIPAGEM PRIMITIVA>

let nome: string = 'andré';
let idade: number = 90;
let programador: boolean = false;

:: Utilizamos esses 3 confirmadores de tipagem, mas não são
exatamente necessarios nesses 3 casos, pois o tipo especifico
já é bem "esclarecido" pelo próprio nome da variavel definido.
<ARRAY>

Numerico ::

let numeros: number[] = [10,12,33,54,78] :: array numerico *recomendado*
let numeros: Array<number> = [10,12,33,54,78] :: array numerico alternativo

String ::

let nomes: string[] = ['pano','pedra','carro'] :: array string *recomendado*
let nomes: Array<string> = ['pano','pedra','carro'] :: array string alternativo

Any ::

let palavras: any = ['pano','pedra','carro'] :: qualquer

palavras.push('566') :: daria um erro, pois o typescript 
detectaria um padrão de string no array palavras, porem a
categoria any diz que ele não deve se preocupar com
a typagem da variavel, logo esse array seria compilado normalmente.
