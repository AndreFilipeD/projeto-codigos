<ANIMATION TRIGGER> 
<CSS>

@keyframes turna{
    0%{transform: rotate(0deg);}
    33%{transform: rotate(40deg);}
    66%{transform: rotate(-20deg);}
    0%{transform: rotate(0deg);}
}
@keyframes turnb{
    0%{transform: rotate(0deg);}
    33%{transform: rotate(40deg);}
    66%{transform: rotate(-20deg);}
    0%{transform: rotate(0deg);}
}
#block{
    background-color: red;
    width: 100px;
    height: 100px;
    animation-duration: 0.4s;
}
<JAVASCRIPT>

var obj = window.document.querySelector("#block")
var turnmode = true

function anim(){
    turnmode = !turnmode /*alterna entre true e false*/
    if(turnmode){
        obj.style="animation-name: turna;"
    }else{
        obj.style="animation-name: turnb;"
    }
}
<HTML>

<div id="block" onclick="anim()">clique aqui para iniciar a animação</div>
